1.StudentDao类

package com.wqm.dao;

import com.wqm.domain.NewStudent;
import com.wqm.domain.Student;
import org.apache.ibatis.annotations.Param;

import java.util.List;

public interface StudentDao {


    /**
     *    一个简单类型的参数:
     *    简单类型:mybatis把java的基本数据类型和String都叫简单类型
     *
     *    在mapper文件中取简单类型的一个参数的值,使用#{任意字符}
     *
     *
     * @param id
     * @return
     */


    public Student selectStudentById(Integer id);


    /**
     *  多个参数
     */


    List<Student> selectMulitParam(@Param("myname") String name,
                                   @Param("myage") Integer age);


    List<Student> selectStudent(Student student);


    int countStudent();



    List<NewStudent> selectNewStudent( );









}

2.StudentMapper.xml

<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.wqm.dao.StudentDao">

              <select id="selectStudentById" parameterType="java.lang.Integer" resultType="com.wqm.domain.Student">
                  select id,name,email,age from student  where id =#{id}

              </select>
    
    
    
    
    <select id="selectMulitParam" resultType="Student">
         select * from student where name = #{myname} or age = #{myage}
        
    </select>


    <select id="selectStudent" resultType="com.wqm.domain.Student">
             select * from student where name = #{name} or age = #{age}

    </select>

    <select id="countStudent" resultType="int">
        select count(*) from student
    </select>

    <resultMap id="NewStudentMap" type="com.wqm.domain.NewStudent">

        <id column="id" property="s_id"/>


        <id column="name" property="s_name"/>
        <id column="email" property="s_email"/>
        <id column="age" property="s_age"/>


    </resultMap>



    <select id="selectNewStudent" resultMap="NewStudentMap">
        select id,name,email,age from student
    </select>



</mapper>

3.测试类

package com.wqm;

import static org.junit.Assert.assertTrue;

import com.wqm.dao.StudentDao;
import com.wqm.domain.NewStudent;
import com.wqm.domain.Student;
import com.wqm.util.MyBatisUtils;
import org.apache.ibatis.session.SqlSession;
import org.junit.Test;

import java.util.List;

/**
 * Unit test for simple App.
 */
public class AppTest 
{
    /**
     * Rigorous Test :-)
     */
    @Test
    public void shouldAnswerWithTrue()
    {
        assertTrue( true );
    }




    @Test
    public void test01(){

        SqlSession sqlSession = MyBatisUtils.getSqlSession();

        StudentDao dao = sqlSession.getMapper(StudentDao.class);


        Student students = dao.selectStudentById(12);

        System.out.println(students);







    }

    @Test
    public void test02(){

        SqlSession sqlSession = MyBatisUtils.getSqlSession();

        StudentDao dao = sqlSession.getMapper(StudentDao.class);


        List<Student> students = dao.selectMulitParam("22",23);


        students.forEach(s -> System.out.println(s));




    }

    @Test
    public void test03(){

        SqlSession sqlSession = MyBatisUtils.getSqlSession();

        StudentDao dao = sqlSession.getMapper(StudentDao.class);

        Student student = new Student();
        student.setName("哈哈");
        student.setAge(23);



        List<Student> students = dao.selectStudent(student);


        students.forEach(s -> System.out.println(s));




    }

    @Test
    public void test04(){

        SqlSession sqlSession = MyBatisUtils.getSqlSession();

        StudentDao dao = sqlSession.getMapper(StudentDao.class);


        int count = dao.countStudent();


       System.out.println("学生数量为："+count);




    }

    @Test
    public void test05(){

        SqlSession sqlSession = MyBatisUtils.getSqlSession();

        StudentDao dao = sqlSession.getMapper(StudentDao.class);


        List<NewStudent> student = dao.selectNewStudent();


        student.forEach(s -> System.out.println(s));


       sqlSession.close();

    }



}
