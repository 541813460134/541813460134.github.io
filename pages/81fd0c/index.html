死锁

提示

说明：本章内容是博主的Java学习笔记，

出现死锁的情况

public class Test10 {

    public static void main(String[] args) {
        SubThread t1 = new SubThread();
        t1.setName("a");
        t1.start();

        SubThread t2 = new SubThread();
        t2.setName("b");
        t2.start();
    }


    static class SubThread extends Thread {
        private static final Object Lock1 = new Object();
        private static final Object Lock2 = new Object();


        public void run() {
            if ("a".equals(Thread.currentThread().getName())) {
                synchronized (Lock1) {
                    System.out.println("a线程获取到Lock1锁，还需要获取Lock2锁");
                    synchronized (Lock2) {
                        System.out.println("a线程获取Lock1锁后又获取到Lock2锁，执行需要执行的代码");
                    }


                }
            }

            if ("b".equals(Thread.currentThread().getName())) {
                synchronized (Lock2) {
                    System.out.println("b线程获取到Lock2锁，还需要获取Lock1锁");
                    synchronized (Lock1) {
                        System.out.println("b线程获取Lock2锁后又获取到Lock1锁，执行需要执行的代码");
                    }


                }

            }


        }
    }

}
